# country random events
on_bi_yearly_pulse = {
	if = {
		limit = {
			NOT = {
				has_global_flag = bi_yearly_happened
			}
		}
		set_global_flag = bi_yearly_happened
		
		country_event = {
			id = magictradegood_events.1
			days = 365
		}
		random_list = {
			10 = {
				trigger = {
					any_province = {
						trade_goods = copper
					}
				}
				random_province = {
					limit = {
						trade_goods = copper
					}
					province_event = {
						id = magictradegood_events.2
					}
				}
			}
		}
	}
}
